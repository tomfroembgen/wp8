%%%%% Introduction to VCD %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
  \frametitle{Vibrational circular dichroism spectroscopy}

	Measurement of the difference in absorption $ \Delta A $ of left and right circularly polarized IR radiation.

	\begin{itemize}
		\item Important technique to analyze chiral molecules
		\item Enantiomers have mirrored spectra
		\item Powerful tool in combination with IR spectroscopy.
	\end{itemize}

	\begin{figure}[bp]
		\centering
		\includegraphics[width=0.8\textwidth]{example_vcd.png}
		\caption{Computed VCD spectra of \textit{(S),(R)}-2-butanol.}%\footfullcite{thomas2016classical}}
	\end{figure}
\end{frame}

%%%%% Why this? %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
  \frametitle{What's new?}

  Computation of VCD spectra so far:
  
  \begin{itemize}
    \item Static quantum chemistry, based on
    \begin{enumerate}
      \item [a)] magnetic field perturbation theory, or
      \item [b)] nuclear velocity perturbation theory
    \end{enumerate}
    \item Easy for gas phase
    \item Inclusion of bulk effects by implicit or explicit solvation models
    \item But: No dynamic effects
  \end{itemize}

  Newly introduced by this paper:

  \begin{itemize}
    \item VCD spectra from \textit{ab initio} MD simulations
    \begin{itemize}
      \item AIMD: Forces are computed by electronic structure method 
    \end{itemize}
    \item Gas and bulk phase
    \item Allows for dynamic effects
  \end{itemize}
\end{frame}

%%%%% Theory %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
  \frametitle{Theory}

  VCD spectrum can be obtained from FT of time cross-correlation function of electric dipole moment $ \mu(t) $ and magnetic dipole moment $ m(t) $.
  %
  \begin{equation}
    \Delta A (\tilde{\nu}) \propto \int
    \left[ \langle \dot{\mu}(\tau) \dot{m}(t+\tau) \rangle_\tau
    - \langle \dot{m}(\tau) \dot{\mu}(t+\tau) \rangle_\tau \right]
    \exp(-2\pi\mathrm{i}c\tilde{\nu}t) \mathrm{d}t
  \end{equation}
  %
  The electric dipole moment is obtained from Voronoi tesselation of the electron density $ \rho $ (fast).
  %
  \begin{equation}
    \mu_k = \int r\cdot\rho(r) \mathrm{d}r~~~k=1,\dots,N 
  \end{equation}
  %
  Key point: The magnetic dipole moment is obtained from the electric current  density $ j $ and the continuity equation (computational bottleneck).
  %
  \begin{equation}
    \frac{\partial \rho(r,t)}{\partial t} + \nabla j(r,t) = 0
  \end{equation}
  %
  \begin{equation}
    m = \frac{1}{2} \int r \cdot j(r) \mathrm{d}r
  \end{equation}
  %
\end{frame}

%%%%% Summary %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\begin{frame}
  \frametitle{Summary}

  VCD spectra from AIMD simulations:

  \begin{itemize}
    \item [+] Accurate
    \item [+] Allow for dynamics
    \item [+] Yield (all) different conformers and their spectra
    \item [+] Separate analysis of every molecule possible
    \item [+] Not restricted to any particular electronic structure method 
    \item [ ]
    \item [$-$] Much higher cost compared to static QC
    \item [$-$] Highly accurate electronic structure methods often not affordable
  \end{itemize}

\end{frame}
